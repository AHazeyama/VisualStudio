104def grad_draw(canvas_w, color_w, w_size):
105 w_width, w_height = w_size
106 mn_r, mn_g, mn_b, mx_r, mx_g, mx_b = color_w
107 max_r = int(mx_r,16) # グラデーション濃 R Hex -> Dec
108 max_g = int(mx_g,16) # G
109 max_b = int(mx_b,16) # B
110 min_r = int(mn_r,16) # グラデーション淡 R
111 min_g = int(mn_g,16) # G
112 min_b = int(mn_b,16) # B
113 line_dec_r = math.floor(((max_r - min_r) / w_height) *100) /100 # グラデーション分解能 R
114 line_dec_g = math.floor(((max_g - min_g) / w_height) *100) /100 # G
115 line_dec_b = math.floor(((max_b - min_b) / w_height) *100) /100 # B
116 # 1Lineずつ描画
117 for i2 in range(0, w_height):
118 val_r = min(255, math.floor(min_r+i2*line_dec_r)) # グラデーション色 R
119 val_g = min(255, math.floor(min_g+i2*line_dec_g)) # G
120 val_b = min(255, math.floor(min_b+i2*line_dec_b)) # B
121 color = f'#%02x%02x%02x' % (val_r, val_g, val_b) # RGBから色生成
122 canvas_w.create_line(0, i2, w_width, i2, fill=color) # 1line描画

147# 再帰処理サブルーチン
148def dwndir(bfr_wd, aft_wd, sps):
149 spspls = sps + " "
150 items = [f.name for f in os.scandir() if not f.name.startswith('.')]
151 dir_wd = os.getcwd()
152 prt_wd = sps + ' => ' + dir_wd + '\n\n'
153 msg.config(state="normal")
154 msg.insert(tk.END, prt_wd, "info")
155 for bfr_nm in items:
156
157 if os.path.isdir(bfr_nm) and bln0.get(): # ディレクトリなら再帰処理(再起指定有の場合)
158 os.chdir(bfr_nm)
159 dwndir(bfr_wd, aft_wd, spspls)
160 os.chdir('..')
161 if "." in bfr_nm: # 拡張子あり
162 w_nam, w_ext = bfr_nm.split(".") # ファイル名分割　変更前名称と拡張子
163 else:
164 w_nam = bfr_nm
165 w_ext = ""
166
167 bfr_wk = "r'" + bfr_wd + "'" # r'^' を変数化
168 aft_wk = "'" + aft_wd + "'" # 'aft_wd'を変数化
169 work_af = "re.sub(" + bfr_wk + ", " + aft_wk + ", " + 'w_nam' + ")" # subコマンド生成
170 aft_nm = os.path.join(dir_wd, eval(work_af) + "." + w_ext) # Path + 名称変更 + 拡張子
171 bfr_nm = os.path.join(dir_wd, bfr_nm) # Path + 変更前名称 + 拡張子
172 prt_wd = spspls + " " + bfr_nm + '\n'
173 prt_wd = prt_wd + spspls + '-> ' + aft_nm + '\n\n'
174 msg.insert(tk.END, prt_wd, "info")
175 os.rename(bfr_nm, aft_nm)
176 prt_wd = sps + ' <= ' + dir_wd + '\n'
177 msg.insert(tk.END, prt_wd, "info")
178 msg.config(state="disabled")
