109 max_b = int(mx_b,16) # B

113 line_dec_r = math.floor(((max_r - min_r) / w_height) *100) /100 # グラデーション分解能 R
114 line_dec_g = math.floor(((max_g - min_g) / w_height) *100) /100 # G

116 # 1Lineずつ描画
