344 def rename_in_directory(self, dir_path: Path, before_word: str, after_word: str, indent: str) -> None:
345 next_indent = indent + " "
346 self.append_message(f"{indent} => {dir_path}\n\n")
347
348 items = [p for p in dir_path.iterdir() if not p.name.startswith(".")]
349 items.sort(key=lambda p: p.name)
350
351 for item in items:
352 # ディレクトリは先に中へ入ってから、最後に自分自身をリネーム
353 if item.is_dir() and self.rec_chk.isChecked():
354 self.rename_in_directory(item, before_word, after_word, next_indent)
355
356 original_path = item
357 stem, suffix = os.path.splitext(item.name)
358
359 if suffix:
360 new_name = re.sub(before_word, after_word, stem) + suffix
361 else:
362 new_name = re.sub(before_word, after_word, item.name)
363
364 new_path = item.with_name(new_name)
365
366 self.append_message(f"{next_indent} {original_path}\n")
367 self.append_message(f"{next_indent}-> {new_path}\n\n")
368
369 if original_path == new_path:
370 continue
371
372 if new_path.exists():
373 raise FileExistsError(str(new_path))
374
375 original_path.rename(new_path)
376 item = new_path
377
378 self.append_message(f"{indent} <= {dir_path}\n")
