26def hash_file(path: Path, algo: Algo) -> str:
27 if algo == "md5":
28 h = hashlib.md5()
29 elif algo == "sha1":
30 h = hashlib.sha1()
31 elif algo == "sha3_256":
32 h = hashlib.sha3_256()
33 elif algo == "sha256":
34 h = hashlib.sha256()
35 elif algo == "sha512":
36 h = hashlib.sha512()
37 elif algo == "blake2b":
38 h = hashlib.blake2b()
39 else:
40 raise ValueError(f"Unknown algo: {algo}")
41
42 with path.open("rb") as f:
43 for chunk in iter(lambda: f.read(1024 * 1024), b""):
44 h.update(chunk)
45 return h.hexdigest()
